#include <stdio.h>
#include <string.h>
#include <unistd.h>

char items[3][10];

void func_a(){
    char buff[10];
     int i = 0;
    for(i=0; i<3; i++){
         printf("input item:");
        fgets(buff, sizeof(buff), stdin);
         buff[strcspn(buff, "\n")] = 0;
        strncpy(items[i], buff, sizeof(items[i])-1);
        items[i][sizeof(items[i])-1] = '\0';
     }
}

void func_b(){
    char *buff;
    printf("done adding items\n");
    int j;
    printf("display item #:");
    scanf("%d", &j);
    buff = &items[j][0];
    printf("item %d: %s\n", j, buff);
}
